-- 查看TiFlash副本状态

WHERE table_schema = 'test' AND table_name = 'orders';

-- 强制使用TiFlash进行查询
SELECT /*+ read_from_storage(tiflash[orders]) */ 
    customer_id, 
    COUNT(*) as order_count,
    SUM(amount) as total_amount
